%&pdflatex
\documentclass{article}

\usepackage{collectbox}

\makeatletter
\newcommand*\Mbox{%
    \leavevmode
    \@ifnextchar\bgroup
        \hbox
        \mbox
}

\newcommand*\Makebox{%
    \@ifnextchar[%
        \@Makebox%
        \Mbox%
}
\def\@Makebox[#1]{%
    \@ifnextchar[%
        {\@@Makebox{#1}}%
        {\@@Makebox{#1}[c]}%
}
\def\@@Makebox#1[#2]{%
    \collectbox*{\@imakebox[{#1}][{#2}]}%
}

\newcommand*\Fbox{\collectbox*\fbox}
\makeatother
